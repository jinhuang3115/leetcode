~~~~//写一个函数，求两个整数之和，要求在函数体内不得使用 “+”、“-”、“*”、“/” 四则运算符号。 
//
// 
//
// 示例: 
//
// 输入: a = 1, b = 1
//输出: 2 
//
// 
//
// 提示： 
//
// 
// a, b 均可能是负数或 0 


//leetcode submit region begin(Prohibit modification and deletion)
/**
 * @param {number} a
 * @param {number} b
 * @return {number}
 */
var add = function(a, b) {
    let c = b;
    while(c){
        let temp = (a ^ c);
        c = (a & c) << 1;
        a = temp;
    }
    return a;
};
